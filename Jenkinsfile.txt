pipeline {
    agent any
    environment {
        TOMCAT_PATH = 'C:/Tomcat/apache-tomcat-9.0.100/webapps/ROOT'  // Updated with the correct path
        TOMCAT_URL = 'http://localhost:8081/manager/text'  // Tomcat Manager URL
    }
    stages {
        stage('Checkout') {
            steps {
                // Checkout the repository to get the project files
                git 'https://github.com/KNSaiShraddha/recipes-blog.gitt'  // Replace with your repository URL
            }
        }
        stage('Deploy to Tomcat') {
            steps {
                script {
                    // Copy the project files to the Tomcat webapps directory
                    bat '''
                    xcopy /E /I * %TOMCAT_PATH%
                    '''
                }
            }
        }
    }
    post {
        always {
            // Optionally, clean up the workspace
            cleanWs()
        }
    }
}
